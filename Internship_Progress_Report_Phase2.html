export const getFeeInsights = async (req, res) => {
  try {
    const totalCollected = await prisma.payment.aggregate({
      _sum: { amount: true }
    });
    res.status(200).json({ success: true, totalCollected });
  } catch (error) {
    res.status(500).json({ error: 'Server Error' });
  }
};
